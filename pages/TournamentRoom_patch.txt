PATCH: Add finalColorCountdown useEffect after winnerCountdown useEffect

// Drive the final color countdown - when it hits 0, move to FINAL_SPIN_COUNTDOWN
useEffect(() => {
  if (finalColorCountdown === null) return;
  if (finalColorCountdown <= 0) {
    setFinalColorCountdown(null);
    setPhase('FINAL_SPIN_COUNTDOWN');
    setCountdown(3);
    setCountdownActive(true);
    return;
  }
  const t = setTimeout(() => setFinalColorCountdown(c => (c !== null ? c - 1 : null)), 1000);
  return () => clearTimeout(t);
}, [finalColorCountdown]);
